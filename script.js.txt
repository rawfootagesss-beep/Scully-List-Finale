const container = document.getElementById('list-container');

function loadLevels(filter = "") {
    container.innerHTML = "";
    levelNames.forEach((name, index) => {
        if (name.toLowerCase().includes(filter.toLowerCase())) {
            const card = document.createElement('div');
            card.className = 'level-card';
            const searchLink = `https://www.youtube.com/results?search_query=Geometry+Dash+Shitty+${name.replace(/ /g, "+")}`;
            
            card.innerHTML = `
                <div class="level-thumb">
                    <img src="https://i.ytimg.com/vi/6d8vIAt6h_s/mqdefault.jpg" alt="thumbnail">
                </div>
                <div class="level-info">
                    <h2>#${index + 1} - ${name}</h2>
                    <p>Creador: Community Project</p>
                    <a href="${searchLink}" target="_blank" style="color:var(--accent); font-size:0.8rem;">Ver Video</a>
                </div>
                <div style="font-size: 1.5rem; color: var(--accent); font-weight: bold;">#${index + 1}</div>
            `;
            container.appendChild(card);
        }
    });
}

function searchLevel() {
    const text = document.getElementById('searchBar').value;
    loadLevels(text);
}

document.addEventListener('DOMContentLoaded', () => loadLevels());
